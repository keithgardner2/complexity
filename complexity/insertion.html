    -(void)insertionSort{
        int j, i, key;
        for(j = 1; j < problemSize; j++){
            key = data[j];
            i = j -1;
            while(i > 0 && data[i] > key){
                data[i + 1] = data[i];
                i = i -1;
            }
            data[i + 1] = key;
        }
    }
